git pull - Atualiza tudo que está no repositório
git pull origin master - baixa/atualiza o repositório
git commit -m "Comentário" - Salva/Altera
git push origin master - Envia pro site

///////////
git status - ve se está tudo certo
git add nomedoarquivo- adiciona arquivos para monitorar (com * adiciona todos)
cd - entra me pasta
ls - ve os arquivos
git rm - remove o arquivo
git clone - baixa o repositório
git checkout nomedoarquivo - Volta versão
